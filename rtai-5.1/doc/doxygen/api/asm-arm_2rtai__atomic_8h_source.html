58 /* Poor man's cmpxchg(). */
59 #define atomic_cmpxchg(p, o, n) ({ \
60  typeof(*(p)) __o = (o); \
61  typeof(*(p)) __n = (n); \
62  typeof(*(p)) __prev; \
63  unsigned long flags; \
64  rtai_hw_lock(flags); \
65  __prev = *(p); \
66  if (__prev == __o) \
67  *(p) = __n; \
68  rtai_hw_unlock(flags); \
69  __prev; })

85 static inline unsigned long
86 atomic_xchg(volatile void *ptr, unsigned long x)
87 {
88  asm volatile(
89  "swp %0, %1, [%2]"
90  : "=&r" (x)
91  : "r" (x), "r" (ptr)
92  : "memory"
93  );
94  return x;
95 }

97 static inline unsigned long atomic_cmpxchg(volatile void *ptr, unsigned long old, unsigned long new)
98 {
99  unsigned long oldval, res;
100 
101  do {
102  __asm__ __volatile__("@ atomic_cmpxchg\n"
103  "ldrex %1, [%2]\n"
104  "teq %1, %3\n"
105  "strexeq %0, %4, [%2]\n"
106  : "=&r" (res), "=&r" (oldval)
107  : "r" (*(unsigned long*)ptr), "r" (old), "r" (new)
108  : "cc");
109  } while (res);
110 
111  return oldval;
112 }

129 static inline int atomic_add_return(int i, atomic_t *v)
130 {
131  unsigned long tmp;
132  int result;
133 
134  __asm__ __volatile__("@ atomic_add_return\n"
135 "1: ldrex %0, [%2]\n"
136 " add %0, %0, %3\n"
137 " strex %1, %0, [%2]\n"
138 " teq %1, #0\n"
139 " bne 1b"
140  : "=&r" (result), "=&r" (tmp)
141  : "r" (&v->counter), "Ir" (i)
142  : "cc");
143 
144  return result;
145 }

147 static inline int atomic_sub_return(int i, atomic_t *v)
148 {
149  unsigned long tmp;
150  int result;
151 
152  __asm__ __volatile__("@ atomic_sub_return\n"
153 "1: ldrex %0, [%2]\n"
154 " sub %0, %0, %3\n"
155 " strex %1, %0, [%2]\n"
156 " teq %1, #0\n"
157 " bne 1b"
158  : "=&r" (result), "=&r" (tmp)
159  : "r" (&v->counter), "Ir" (i)
160  : "cc");
161 
162  return result;
163 }
