40 static inline int rtai_timer_irq_ack( void )
41 {
42  OSSR = OSSR_M0;
43  rt_unmask_irq(RTAI_TIMER_IRQ);
44  if ( (int)(OSCR - OSMR0) < 0 )
45  /* This is what just happened: we were setting a next timer interrupt
46  in the scheduler, while a previously configured timer-interrupt
47  happened, so, just restore the correct value of the match register and
48  proceed as usual. Yes, we will have to re-calculate the next interrupt. */
49  OSMR0 = OSCR - 1;
50  return 0;
51 }

59 static inline RTIME rtai_rdtsc(void)
60 {
61  RTIME ts;
62  unsigned long flags, count;
63 
64  rtai_save_flags_and_cli(flags);
65 
66  if ( ( count = OSCR ) < rtai_tsc.hltsc[0] )
67  rtai_tsc.hltsc[1]++;
68  rtai_tsc.hltsc[0] = count;
69  ts = rtai_tsc.tsc;
70  rtai_restore_flags(flags);
71 
72  return ts;
73 }

79 static inline void rt_set_timer_match_reg(int delay)
80 {
81  unsigned long flags;
82  unsigned long next_match;
83 
84 #ifdef PROTECT_TIMER
85  if ( delay > LATCH )
86  delay = LATCH;
87 #endif
88 
89  rtai_save_flags_and_cli(flags);
90 
91  if ( delay )
92  next_match = OSMR0 = delay + OSCR;
93 // next_match = OSMR0 = delay + rtai_tsc.hltsc[0];
94  else
95  next_match = ( OSMR0 += rt_times.periodic_tick );
96 // next_match = OSMR0 = ((unsigned long *)&rt_times.intr_time)[0];
97 
98 #ifdef PROTECT_TIMER
99  while ((int)(next_match - OSCR) < 2 * SETUP_TIME_TICKS ) {
100  OSSR = OSSR_M0; /* Clear match on timer 0 */
101  next_match = OSMR0 = OSCR + 4 * SETUP_TIME_TICKS;
102  }
103 #endif
104 
105  rtai_restore_flags(flags);
106 }
107 #endif
