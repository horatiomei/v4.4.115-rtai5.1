361 {

364 
365  if ((arg.tasklet = (struct rt_tasklet_struct*)rtai_lxrt(TASKLETS_IDX, SIZARG, INIT, &arg).v[LOW])) {
366  if ((is_hard = rt_is_hard_real_time(NULL))) {
367  rt_make_soft_real_time();

375  nanosleep(&delay, NULL);
376  }
