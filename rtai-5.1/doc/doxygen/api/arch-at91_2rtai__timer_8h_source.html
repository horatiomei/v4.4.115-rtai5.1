44 static inline unsigned int at91_tc_read(unsigned int reg_offset)
45 {
46  unsigned long addr =
47  (AT91_VA_BASE_TCB0 + 0x40 * CONFIG_IPIPE_AT91_TC);
48 
49  return readl((void __iomem *)(addr + reg_offset));
50 }
51 
52 static inline void at91_tc_write(unsigned int reg_offset, unsigned long value)
53 {
54  unsigned long addr =
55  (AT91_VA_BASE_TCB0 + 0x40 * CONFIG_IPIPE_AT91_TC);
56 
57  writel(value, (void __iomem *)(addr + reg_offset));
58 }

70 static inline RTIME rtai_rdtsc(void)
71 {
72  if(!rt_periodic)
73  /*
74  * one-shot mode : use ipipe native get tsc
75  */
76  return __ipipe_mach_get_tsc();
77  else
78  /*
79  * periodic mode : use specific get tsc
80  * tsc has been updated since last shot by rtai_at91_update_tsc
81  */
82  return __ipipe_get_tsc()+at91_tc_read(AT91_TC_CV);
83 }

91  __ipipe_mach_set_dec(delay);

93  /*
94  * periodic mode: at91's TC reload itself
95  * so nothing to do
96  */
97  }
98 }
