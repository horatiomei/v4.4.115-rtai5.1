183 #ifndef rtai_llimd
184 /* Division of an unsigned 96 bits ((h << 32) + l) by an unsigned 32 bits.
185  Building block for llimd. Without const qualifiers, gcc reload registers
186  after each call to uldivrem. */
187 static inline unsigned long long
188 __rtai_generic_div96by32 (const unsigned long long h,
189  const unsigned l,
190  const unsigned d,
191  unsigned long *const rp)
192 {
193  unsigned long rh;
194  const unsigned qh = rtai_uldivrem(h, d, &rh);
195  const unsigned long long t = __rtai_u64fromu32(rh, l);
196  const unsigned ql = rtai_uldivrem(t, d, rp);
197 
198  return __rtai_u64fromu32(qh, ql);
199 }

201 extern inline unsigned long long
202 __rtai_generic_ullimd (const unsigned long long op,
203  const unsigned m,
204  const unsigned d)
205 {
206  unsigned oph, opl, tlh, tll;
207  unsigned long long th, tl;
208 
209  __rtai_u64tou32(op, oph, opl);
210  tl = rtai_ullmul(opl, m);
211  __rtai_u64tou32(tl, tlh, tll);
212  th = rtai_ullmul(oph, m);
213  th += tlh;
214 
215  return __rtai_generic_div96by32(th, tll, d, NULL);
216 }

278 {
279  struct list_head *p = hal_pipeline.next;
