57 /*
58  * Registers according to the ARM procedure call standard:
59  * Reg Description
60  * r0-r3 argument/scratch registers
61  * r4-r9 variable register
62  * r10=sl stack limit/variable register
63  * r11=fp frame pointer/variable register
64  * r12=ip intra-procedure-call scratch register
65  * r13=sp stack pointer (auto preserved)
66  * r14=lr link register
67  * r15=pc program counter (auto preserved)
68  *
69  * Take special care to restore r0/r1 for startup of new task (->
70  * init_arch_stack, rt_startup)!
71  *
72  * Stack layout:
73  *
74  * Offset Content
75  *
76  * 12 r14=lr (i.e. resume point for task)
77  * 11 r11=fp
78  * 10 r10=sl
79  * 9 r9
80  * 8 r8
81  * 7 r7
82  * 6 r6
83  * 5 r5
84  * 4 r4
85  * 3 cpsr
86  * 2 domain_access_control
87  * 1 r1 (startup: data)
88  * 0 r0 (startup: rt_thread)
89  */
90 
91 #define rt_exchange_tasks(old_task, new_task) \
92  asm volatile( \
93  "mrc p15, 0, r2, c3, c0\n\t" /* get current domain_access_control */ \
94  "adr lr, 1f\n\t" /* get address of resume point */ \
95  "mrs r3, cpsr\n\t" /* get current cpsr */ \
96  "ldr ip, [%[oldp]]\n\t" /* get pointer to old task */ \
97  "stmfd sp!, {r0 - fp, lr}\n\t" /* push registers on stack */ \
98  "str %[new], [%[oldp]]\n\t" /* *oldp = new */ \
99  "str sp, [ip]\n\t" /* save current stack-pointer to old task */ \
100  "ldr sp, [%[new]]\n\t" /* get stack-pointer of new task */ \
101  "ldmfd sp!, {r0-r3}\n\t" /* pop new tasks' r0, r1, cpsr & d.a.c from stack */ \
102  "mcr p15, 0, r2, c3, c0\n\t" /* restore previous domain_access_control */ \
103  "msr cpsr_c, %[psr]\n\t" /* disable hw-irqs */ \
104  "msr spsr, r3\n\t" /* set spsr to previous cpsr */ \
105  "ldmfd sp!, {r4 - fp, pc}^\n\t" /* pop registers, pc = resume point, cpsr = spsr */ \
106  "1:" /* resume point (except for startup) */ \
107  : /* output */ /* none */ \
108  : /* input */ [oldp] "r" (&old_task), \
109  [new] "r" (new_task), \
110  [psr] "i" (SVC_MODE|I_BIT) \
111  : /* clobber */ "lr", "ip", "r2", "r3", "memory" \
112  )

114 extern inline unsigned long
115 current_domain_access_control(void)
116 {
117  unsigned long domain_access_control;
118  asm("mrc p15, 0, %0, c3, c0" : "=r" (domain_access_control));
119  return domain_access_control;
120 }

122 /*
123  * Set up stack for new proper RTAI task (i.e. kernel task without
124  * kernel-thread). This is called in rt_task_init_cpuid() where the following
125  * variables are used (should be arguments to macro!):
126  * rt_thread pointer to tasks function
127  * data integer to pass to rt_thread
128  *
129  * The stack needs to be set-up so that a rt_exchange_tasks() on the newly
130  * created task will call rt_startup(rt_thread = r0, data = r1).
131  *
132  * See rt_exchange_tasks() above for the stack layout.
133  *
134  */
135 #define init_arch_stack() \
136  do { \
137  task->stack -= 13; /* make room on stack */ \
138  task->stack[12] = (int)rt_startup; /* entry point */ \
139  task->stack[ 3] = SVC_MODE; /* cpsr */ \
140  task->stack[ 2] = (int)current_domain_access_control(); \
141  task->stack[ 1] = (int)data; /* arg 2 of rt_startup() */ \
142  task->stack[ 0] = (int)rt_thread; /* arg 1 " " */ \
143  } while (0)
