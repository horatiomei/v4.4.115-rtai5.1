210 RTAI_PROTO (void *, _rt_shm_alloc, (void *start, unsigned long name, int size, int suprt, int isheap))
211 {
212  int hook;
213  void *adr = NULL;
214 
215  if ((hook = open(RTAI_SHM_DEV, O_RDWR)) <= 0) {
216  return NULL;
217  } else {
218  struct { unsigned long name; long arg, suprt; } arg = { name, size, suprt };
219 #ifdef SHM_USE_LXRT
220  if ((size = rtai_lxrt(BIDX, SIZARG, SHM_ALLOC, &arg).i[LOW])) {
221 #else
222  if ((size = ioctl(hook, SHM_ALLOC, (unsigned long)(&arg)))) {
223 #endif
224  if ((adr = mmap(start, size, PROT_WRITE | PROT_READ, MAP_SHARED | MAP_LOCKED, hook, 0)) == MAP_FAILED) {;
225 #ifdef SHM_USE_LXRT
226  rtai_lxrt(BIDX, sizeof(name), SHM_FREE, &name);
227 #else
228  ioctl(hook, SHM_FREE, &name);
229 #endif
230  } else if (isheap) {
231  arg.arg = (unsigned long)adr;
232 #ifdef SHM_USE_LXRT
233  rtai_lxrt(BIDX, SIZARG, HEAP_SET, &arg);
234 #else
235  ioctl(hook, HEAP_SET, &arg);
236 #endif
237  }
238  }
239  }
240  close(hook);
241  return adr;
242 }

352 RTAI_PROTO(int, rt_shm_free, (unsigned long name))
353 {
354  int hook, size;
355  struct { void *nameadr; } arg = { &name };
356  if ((hook = open(RTAI_SHM_DEV, O_RDWR)) <= 0) {
357  return 0;
358  }
359 // no SHM_FREE needed, we release it all and munmap will do it through
360 // the vma close operation provided by shm.c
361 #ifdef SHM_USE_LXRT
362  if ((size = rtai_lxrt(BIDX, SIZARG, SHM_SIZE, &arg).i[LOW])) {
363 #else
364  if ((size = ioctl(hook, SHM_SIZE, (unsigned long)&arg))) {
365 #endif
366  if (munmap((void *)name, size)) {
367  size = 0;
368  }
369  }
370  close(hook);
371  return size;
372 }
