82 /* TRACE_RTAI_GLOBAL_IRQ_ENTRY */
83 typedef struct _trace_rtai_global_irq_entry
84 {
85  uint8_t irq_id; /* IRQ number */
86  uint8_t kernel; /* Are we executing kernel code */
87 } LTT_PACKED_STRUCT trace_rtai_global_irq_entry;
88 #if CONFIG_X86
89 #define TRACE_RTAI_GLOBAL_IRQ_ENTRY(ID, __dummy) \
90  do \
91  {\
92  uint32_t eflags, xcs; \
93  trace_rtai_global_irq_entry irq_entry;\
94  irq_entry.irq_id = ID;\
95  __asm__ __volatile__("pushfl; pop %0": "=g" (eflags)); \
96  __asm__ __volatile__("pushl %%cs; pop %0": "=g" (xcs)); \
97  irq_entry.kernel = !((VM_MASK & eflags) || (3 & xcs));\
98  rt_trace_event(TRACE_RTAI_EV_GLOBAL_IRQ_ENTRY, &irq_entry);\
99  } while(0)
100 #endif
101 #if CONFIG_PPC
102 #define TRACE_RTAI_GLOBAL_IRQ_ENTRY(ID, KERNEL) \
103  do \
104  {\
105  trace_rtai_global_irq_entry irq_entry;\
106  irq_entry.irq_id = ID;\
107  irq_entry.kernel = KERNEL;\
108  rt_trace_event(TRACE_RTAI_EV_GLOBAL_IRQ_ENTRY, &irq_entry);\
109  } while(0)
110 #endif
