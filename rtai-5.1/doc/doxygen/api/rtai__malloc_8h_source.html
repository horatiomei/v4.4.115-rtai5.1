37 /*
38  * LIMITS:
39  *
40  * Minimum page size is 2 ** RTHEAP_MINLOG2 (must be large enough to
41  * hold a pointer).
42  *
43  * Maximum page size is 2 ** RTHEAP_MAXLOG2.
44  *
45  * Minimum block size equals the minimum page size.
46  *
47  * Requested block size smaller than the minimum block size is
48  * rounded to the minimum block size.
49  *
50  * Requested block size larger than 2 times the page size is rounded
51  * to the next page boundary and obtained from the free page
52  * list. So we need a bucket for each power of two between
53  * RTHEAP_MINLOG2 and RTHEAP_MAXLOG2 inclusive, plus one to honor
54  * requests ranging from the maximum page size to twice this size.
55  */
56 
57 #define RTHEAP_MINLOG2 4
58 #define RTHEAP_MAXLOG2 22
59 #define RTHEAP_MINALLOCSZ (1 << RTHEAP_MINLOG2)
60 #define RTHEAP_MINALIGNSZ RTHEAP_MINALLOCSZ
61 #define RTHEAP_NBUCKETS (RTHEAP_MAXLOG2 - RTHEAP_MINLOG2 + 2)
